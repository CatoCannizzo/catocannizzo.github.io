const int analogIn = A0;
const int LEDout = 6;
int light = 0;
int lightMapped = 0;
void setup() {
    // put your setup code here, to run once:
    Serial.begin(9600);
}

void loop() {
    // put your main code here, to run repeatedly:
    light = analogRead(analogIn);
    lightMapped = map(constrain(light, 200, 580), 200, 580, 0, 255);
    analogWrite(LEDout, lightMapped);
    Serial.print("light = ");
    Serial.println(light);
    
}
