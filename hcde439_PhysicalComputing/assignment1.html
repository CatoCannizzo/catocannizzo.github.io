void loop() {
    // put your main code here, to run repeatedly:

    // A for loop to run the following segment 3 times, telling me my code is about to start
    for (int i = 0; i <= 3; i++) {
        // turns on led connected to pin 1
        digitalWrite(1, HIGH);
        // turns on led connected to pin 2
        digitalWrite(2, HIGH);
        // turns on led connected to pin 3
        digitalWrite(3, HIGH);
        // stops code for wait1 miliseconds
        delay(wait1);
        // turns off led connected to pin 1
        digitalWrite(1, LOW);
        // turns off led connected to pin 2
        digitalWrite(2, LOW);
        // turns off led connected to pin 3
        digitalWrite(3, LOW);
        // stops code for wait1 miliseconds
        delay(wait1);
    }
    // waits 0.75 sec for lights to start
    delay(750);
    // Calls main function of repeating code
    melodyFunction();
    // Calls main function of repeating code
    melodyFunction();
    // Calls secondary segment of code passing in 3 to light up the 3 LED
    oneUhPenny(3);
    // Calls secondary segment of code passing in 2 to light up the 2 LED
    oneUhPenny(2);
    // Calls main function of repeating code
    melodyFunction();
    // waits 2 seconds to give visual rest after the presentation
    delay(2000);  
}

// makes a function for the secondary repeated segment of code with a variable being passed in for which LED to light up
int oneUhPenny(int x) {
    // For loop to run the following segment of code 4 times
    for (int i=0; i <=4; i++){
        //Turns on the LED (pin) for which ever variable was passed
        digitalWrite(x, HIGH);
        // stops code for wait4 miliseconds
        delay(wait4);
        //Turns off the LED (pin) for which ever variable was passed
        digitalWrite(x, LOW);
        // stops code for wait1 miliseconds
        delay(wait1);
    }
}
